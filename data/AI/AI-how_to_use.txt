(5) 怎么做到在Flask中调用DeepSeek时，也保持流式输出呢？

from flask import Flask, Response, stream_with_context
import requests

app = Flask(__name__)

# DeepSeek API 的 URL 和 API Key
DEEPSEEK_API_URL = "http://127.0.0.1:11434/v1/chat/completions"
API_KEY = "sk-test2"

# 调用 DeepSeek API 并流式返回数据
def call_deepseek_stream(user_model, user_msg, user_stream):
    headers = {
        "Authorization": f"Bearer {API_KEY}",
        "Content-Type": "application/json"
    }
    data = {
        "model": user_model,
        "messages": user_msg,
        "stream": user_stream
    }
    # 发送流式请求
    response = requests.post(DEEPSEEK_API_URL, headers=headers, json=data, stream=True)
    # 逐步返回 DeepSeek 的响应数据
    for chunk in response.iter_content(chunk_size=None):
        if chunk:
            yield chunk

# Flask 路由，调用 DeepSeek 并流式返回结果
@app.route('/deepseek-stream')
def deepseek_stream():
    prompt = "你好啊"  # 替换为你的提示词
    # 使用 stream_with_context 包装流式调用
    return Response(stream_with_context(call_deepseek_stream(prompt)), content_type='text/plain')

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=8089, debug=True)

from flask import Flask, request, jsonify, Response, stream_with_context

# v0.5 支持查询 model 名字; 验证API key是否合法; 合法才能对话
# v0.5.1 支持IP和API key匹配检查


# DeepSeek API direct URL
base_URL = "http://127.0.0.1:11434"
API_keys_file = 'deepseek_gate_API_kyes.csv'

# load API key info
def load_api_keys():
    api_keys = {}
    with open(API_keys_file, mode='r') as file:
        reader = csv.reader(file)
        for row in reader:
            api_keys[row[0]] = [row[1], row[2]]
    return api_keys



# 1. get model list
@app.route('/deepseek/models', methods=['GET'])
def getModels():
    headers = {
        #"Authorization": "Bearer sk-test2",
        "Content-Type": "application/json"
    }
    response = requests.get(f"{base_URL}/v1/models", headers=headers)

    if response.status_code == 200:
        json_response = response.json()
    else:
        json_response={"Error":"get model list error from deepseek"}

    return jsonify(json_response), response.status_code




# chat with deepseek: real work
def call_deepseek_stream(raw_data):
    headers = {
        "Authorization": "Bearer API_key_here", #这直接传递给deepseek，它不做验证
        "Content-Type": "application/json"
    }
    json_data = json.loads(raw_data)
    # 发送流式请求
    url=f"{base_URL}/v1/chat/completions"
    response = requests.post(url, headers=headers, json=json_data, stream=True)
    # 逐步返回 DeepSeek 的响应数据
    for chunk in response.iter_content(chunk_size=None):
        if chunk:
            yield chunk

# get IP from Nginx
def get_client_ip():
    # 优先检查 X-Forwarded-For
    if request.headers.getlist("X-Forwarded-For"):
        ip = request.headers.getlist("X-Forwarded-For")[0]
    # 其次检查 X-Real-IP
    elif request.headers.get("X-Real-IP"):
        ip = request.headers.get("X-Real-IP")
    # 最后使用 remote_addr
    else:
        ip = request.remote_addr
    return ip

# chat with deepseek, if API key client is valid and match user's IP.
@app.route('/deepseek/chat/completions', methods=['POST'])
def chat():
    print("[0]>> begin to chat")
    api_keys = load_api_keys() #get all API keys from csv file.
    api_key = request.headers.get('Authorization').split(" ")[1] #get user's key
    if api_key in api_keys:
        # check IP
        IP = get_client_ip()
        print("[1]>> user's IP", IP)
        if IP != api_keys[api_key][1]:
            return jsonify({"status":"error", "error":{"code":"invalid_api_key", "message": "API key and IP NOT match."}}), 403

        # check date
        expiry_date = datetime.datetime.strptime(api_keys[api_key][0], "%Y/%m/%d")
        if expiry_date >= datetime.datetime.now():
            # call DeepSeek service as if calling the service directly
            post_data=request.get_data()
            return Response(stream_with_context(call_deepseek_stream(post_data)), content_type='text/plain')
        else:
            return jsonify({"status":"error", "error":{"code":"invalid_api_key", "message": "API key has expired."}}), 403
    else:
        return jsonify({"status":"error", "error":{"code":"invalid_api_key", "message":"Invalid API key."}}), 403

        # 转发请求到DeepSeek-R1模型的API

        proxy_set_header Connection $http_connection;
    }
}

	日期格式：确保日期格式为 YYYY/MM/DD，并且日期有效。

ChatGPT 经验：要把需求不断拆分细化，达到具体函数的颗粒度！AI的智商上限=你的表达能力
----------------------------------------
1.要拆分的足够细：这就要求人要掌握，否则无法驾驭

ChatGPT到4的时候已经很好用了，大幅加快了开发效率，但要认清目前AI能力的边界，你不可能整个项目丢给它，要把需求不断拆分细化，达到具体函数的颗粒度，描述好函数功能，输入的参数，需要的结果，大部分都可以帮你快速搞定，当然也会有胡说八道的情况，但越来越少了，最近已经不续费ChatGPT plus了，用DeepSeek基本够了。


-> 这就要求先系统性学习，全面掌握。后来忘了细节，然后让AI帮忙快速推进。



2.怎么具体呢？
记住反智障公式：
- 垃圾："帮我写一个X"
- 人话："你是[什么身份]，在[什么场景]，给[哪个冤种]，写个[什么效果]的X"

举例：
- 智障："写个离职信"
- 正常："你是被压榨两年的社畜，老板还欠三个月工资，写封离职信让老板失眠"

这个区别，大到你妈都想打你。

(2)我给你看个现场处刑
AI杀手版："帮我写一个邮件"
AI输出："尊敬的领导，您好，祝您工作顺利……"

这种垃圾我闭着眼都能写。

核武版："你是被甲方折磨三个月的乙方，今天17:58客户发来第47版需求变更，写封邮件告诉他要加钱，专业但要让他感受到杀气。"
AI输出："王总，您好。已收到您17:58发送的第47版需求变更。新增的23个功能点已超出原合同范围，根据补充协议，超范围需求按加急项目计费。详细报价见附件，烦请今日内确认。特别提醒：明天为财务结算日。"

看到了吗？这就是AI杀手和正常人的区别。
