C++ 构造函数会自动执行。

#include<iostream>

}

1. 脚本内容
/* 获取命令行传过来的参数 */
#include<iostream>
using namespace std;

string ename[2],iname[2],fname[2],oname;

// 解析命令行参数，共7个参数，并打印出来内容
/*
$ g++ b3_get_cmd_params.cpp
$ ./a.out exon_R1 exon_R2 intronR1 intronR2 fq1 fq2 outerOvere
*/
int main(int narg, char** args){
  // 参数个数 narg
  // 字符串 args
  for(int w=0; w<narg; w++){
    cout << w << ": " << args[w] << endl;
  }
  printf("========");

  int i;
  string *sargs;
  
  int argscnt = 0;
  bool wrong_args = false;
  sargs = new string[narg]; 

  for(i=0;i<narg-1;i++){
    sargs[i] = string(args[i+1]);
  }

  for(i=0;i<narg-1;i++){
    if(argscnt == 0){
      ename[0] = sargs[i];
    }else if(argscnt == 1){
      ename[1] = sargs[i];
    }else if(argscnt == 2){
      iname[0] = sargs[i];
    }else if(argscnt == 3){
      iname[1] = sargs[i];
    }else if(argscnt == 4){
      fname[0] = sargs[i];
    }else if(argscnt == 5){
      fname[1] = sargs[i];
    }else if(argscnt == 6){
      oname = sargs[i];
    }
    argscnt++;    
  }

  if (argscnt != 7){
    cout << "Usage:" << endl;
    cout << args[0] << " sam_file_exon_R1 sam_file_exon_R2 sam_file_intron_R1 sam_file_intron_R2 fastq_file_R1 fastq_file_R2 output" << endl;  
    exit(0);
  }

  cout << "exon:" << ename[0] << ", " << ename[1] << endl;
  cout << "intron:" << iname[0] << ", " << iname[1] << endl;
  cout << "fname:" << fname[0] << ", " << fname[1] << endl;
  cout << "oname:" << oname << endl;

  return 0;
}

总结：
	main 函数的2个参数用于接收命令行参数：main(int narg, char** args)
		参数1：参数个数
		参数2：指针，指向第一个参数字符串
	参数从下标0开始，表示脚本本身的名字。
	然后是每个参数，从1开始，这里共7个。

/* 读取文件，并用 tab 键分隔每行，返回第0个 */
#include<iostream>
#include <fstream>
using namespace std;

string ename;

/*
$ g++ b4_read_file.cpp
$ ./a.out ReadMe.txt
*/

// 对一行字符串，获取tab分隔的第0个元素
string GetWordtab(int &n,string ssline)
{
    string p;
    int n2 = n;
    int s = ssline.size();
    // 到 \t 停止
    while(n < s &&
            ssline[n] != '\t'){    
        n++;
    }
    //截取位置
    if(n-n2 > 0)
        p = ssline.substr(n2,n-n2);
    // 移动n到下一个位置
    n++;
    return p;
}

int main(int argn, char** args){
    int i=0, n;
    ifstream exi;
    ename=args[1];
    string sline, word;

    // 打开文件流
    exi.open(ename.c_str(), ios::in);
    if(!exi){
        cout << "Couldn't open " << ename << endl;
        exit(-1);
    }

    // 读取文件内容
    while(getline(exi, sline)){
        i++;
        if(sline[0] == '@') //跳过 @开头的行
            continue;
        //cout << i << ": " << sline << endl;
        // 获取tab分隔的第0个
        n = 0;
		word = GetWordtab(n,sline);  //提取tab分隔的单词，要第一个
        cout << n << "|" << word << "|" << endl;
    }

    // 关闭文件流
    exi.close();
    cout << "读取文件结束" << endl;
    return 0;
}
