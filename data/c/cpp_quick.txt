/* 读取文件，并用 tab 键分隔每行，返回第0个 */
#include<iostream>
#include <fstream>
using namespace std;

string ename;

/*
$ g++ b4_read_file.cpp
$ ./a.out ReadMe.txt
*/

// 对一行字符串，获取tab分隔的第0个元素
string GetWordtab(int &n,string ssline)
{
    string p;
    int n2 = n;
    int s = ssline.size();
    // 到 \t 停止
    while(n < s &&
            ssline[n] != '\t'){    
        n++;
    }
    //截取位置
    if(n-n2 > 0)
        p = ssline.substr(n2,n-n2);
    // 移动n到下一个位置
    n++;
    return p;
}

int main(int argn, char** args){
    int i=0, n;
    ifstream exi;
    ename=args[1];
    string sline, word;

    // 打开文件流
    exi.open(ename.c_str(), ios::in);
    if(!exi){
        cout << "Couldn't open " << ename << endl;
        exit(-1);
    }

    // 读取文件内容
    while(getline(exi, sline)){
        i++;
        if(sline[0] == '@') //跳过 @开头的行
            continue;
        //cout << i << ": " << sline << endl;
        // 获取tab分隔的第0个
        n = 0;
		word = GetWordtab(n,sline);  //提取tab分隔的单词，要第一个
        cout << n << "|" << word << "|" << endl;
    }

    // 关闭文件流
    exi.close();
    cout << "读取文件结束" << endl;
    return 0;
}

========================================
c++内存管理 - 节点释放/销毁: malloc / calloc 和 free 配对(C风格的内存分配)，new / new[] 和 delete / delete[] 配对 (C++ 动态内存管理)
----------------------------------------
1.单节点释放

// 如果用 new 分配
TreeNode* node = new TreeNode{...};
delete node;  // 正确释放，会调用析构函数

// 如果用 malloc 分配：
TreeNode* node = (TreeNode*)malloc(sizeof(TreeNode));
free(node);    // 仅释放内存，不调用析构函数（可能导致 string 泄漏）



2. 递归释放整棵二叉树

void freeTree(TreeNode* root) {
    if (root == nullptr) return;
    freeTree(root->left);   // 递归释放左子树
    freeTree(root->right);  // 递归释放右子树
    delete root;            // 释放当前节点（若用 new 分配）
}




3. 示例

(1) c++ 推荐使用new: 将 malloc 替换为 new
// 原代码：
TreeNode* root = (TreeNode*)malloc(sizeof(TreeNode));
// 修改后：
TreeNode* root = new TreeNode{stu1, nullptr, nullptr};  // 初始化指针为 nullptr


(2) 释放时统一用 delete：
// 程序结束前释放整棵树
freeTree(root);  // 内部使用 delete




4.最佳实践建议
(1)优先使用 new/delete：确保对象构造/析构正确执行（尤其含 string 等非POD成员时）。
(2)避免混用：不要用 free 释放 new 分配的内存，反之亦然。
(3)使用智能指针（C++11+）：
#include <memory>
std::unique_ptr<TreeNode> root(new TreeNode{...});
// 自动释放内存，无需手动 delete

修正后的代码将不再有内存泄漏风险，且能正确处理 string 成员的析构。

3.在linux下，能精确控制占满n个CPU核心，并且过几秒会随机切换CPU核心

$ cat cpu_controller.cpp

#include <iostream>
#include <thread>
#include <atomic>
#include <vector>
#include <random>
#include <chrono>
#include <sched.h>
#include <unistd.h>
#include <algorithm>
#include <iomanip>

std::atomic<bool> keepRunning(true);

void pinToCoreAndWork(int coreId, int durationSec) {
    cpu_set_t cpuset;
    CPU_ZERO(&cpuset);
    CPU_SET(coreId, &cpuset);
    pthread_setaffinity_np(pthread_self(), sizeof(cpu_set_t), &cpuset);

    auto endTime = std::chrono::steady_clock::now() + 
                  std::chrono::seconds(durationSec);
    while (keepRunning && std::chrono::steady_clock::now() < endTime) {
        asm volatile("" ::: "memory"); // 防止空循环被优化
    }
}

int main(int argc, char** argv) {
    if (argc < 3) {
        std::cerr << "Usage: " << argv[0] 
                  << " <cpu_usage_ratio> <switch_interval_seconds>\n";
        return 1;
    }

    const double usageRatio = std::stod(argv[1]);
    const int switchInterval = std::stoi(argv[2]);
    const int totalCores = sysconf(_SC_NPROCESSORS_ONLN);
    const int targetCores = std::max(1, static_cast<int>(totalCores * usageRatio));

    std::random_device rd;
    std::mt19937 gen(rd());
    std::uniform_int_distribution<> dist(0, totalCores-1);

    while (keepRunning) {
        std::vector<std::thread> workers;
        std::vector<int> selectedCores;

        // 随机选择不重复的核心
        while (selectedCores.size() < targetCores) {
            int core = dist(gen);
            if (std::find(selectedCores.begin(), 
                         selectedCores.end(), core) == selectedCores.end()) {
                selectedCores.push_back(core);
                workers.emplace_back(pinToCoreAndWork, core, switchInterval);
            }
        }

        // 显示当前占用的核心
        auto now = std::chrono::system_clock::now();
		auto now_time = std::chrono::system_clock::to_time_t(now);
		std::tm* tm_now = std::localtime(&now_time);
		std::cout << "[" << std::put_time(tm_now, "%Y-%m-%d %H:%M:%S") 
				  << "] Using cores:";
        for (int core : selectedCores) std::cout << " " << core;
        std::cout << " for " << switchInterval << " seconds\n";

        // 等待切换周期
        std::this_thread::sleep_for(std::chrono::seconds(switchInterval));

        // 清理当前批次的线程
        for (auto& t : workers) {
            if (t.joinable()) t.join();
        }
    }
    return 0;
}

#include <iostream>
#include <vector>
#include <cstdlib>

class MemoryAllocator {
public:
    explicit MemoryAllocator(size_t gb) : size_(gb * 1024UL * 1024 * 1024) {}
    
    void allocate() {
        memory_.resize(size_ / sizeof(int));
        for (auto& block : memory_) {
            block = 1; // 确保内存被实际使用
        }
    }

private:
    std::vector<int> memory_;
    const size_t size_;
};

int main(int argc, char* argv[]) {
    if (argc < 2) {
        std::cerr << "Usage: " << argv[0] << " <GB>" << std::endl;
        return 1;
    }
    size_t gb = std::stoul(argv[1]);
    MemoryAllocator allocator(gb);
    allocator.allocate();
    
	std::cout << "Press Enter to stop..." << std::endl;
    std::cin.ignore();
    return 0;
}
