/* 获取命令行传过来的参数 */
#include<iostream>
using namespace std;

string ename[2],iname[2],fname[2],oname;

// 解析命令行参数，共7个参数，并打印出来内容
/*
$ g++ b3_get_cmd_params.cpp
$ ./a.out exon_R1 exon_R2 intronR1 intronR2 fq1 fq2 outerOvere
*/
int main(int narg, char** args){
  // 参数个数 narg
  // 字符串 args
  for(int w=0; w<narg; w++){
    cout << w << ": " << args[w] << endl;
  }
  printf("========");

  int i;
  string *sargs;
  
  int argscnt = 0;
  bool wrong_args = false;
  sargs = new string[narg]; 

  for(i=0;i<narg-1;i++){
    sargs[i] = string(args[i+1]);
  }

  for(i=0;i<narg-1;i++){
    if(argscnt == 0){
      ename[0] = sargs[i];
    }else if(argscnt == 1){
      ename[1] = sargs[i];
    }else if(argscnt == 2){
      iname[0] = sargs[i];
    }else if(argscnt == 3){
      iname[1] = sargs[i];
    }else if(argscnt == 4){
      fname[0] = sargs[i];
    }else if(argscnt == 5){
      fname[1] = sargs[i];
    }else if(argscnt == 6){
      oname = sargs[i];
    }
    argscnt++;    
  }

  if (argscnt != 7){
    cout << "Usage:" << endl;
    cout << args[0] << " sam_file_exon_R1 sam_file_exon_R2 sam_file_intron_R1 sam_file_intron_R2 fastq_file_R1 fastq_file_R2 output" << endl;  
    exit(0);
  }

  cout << "exon:" << ename[0] << ", " << ename[1] << endl;
  cout << "intron:" << iname[0] << ", " << iname[1] << endl;
  cout << "fname:" << fname[0] << ", " << fname[1] << endl;
  cout << "oname:" << oname << endl;

  return 0;
}

----------------------------------------
1. 打开文件
/* 读取文件，并用 tab 键分隔每行，返回第0个 */
#include<iostream>
#include <fstream>
using namespace std;

string ename;

/*
$ g++ b4_read_file.cpp
$ ./a.out ReadMe.txt
*/

// 对一行字符串，获取tab分隔的第0个元素
string GetWordtab(int &n,string ssline)
{
    string p;
    int n2 = n;
    int s = ssline.size();
    // 到 \t 停止
    while(n < s &&
            ssline[n] != '\t'){    
        n++;
    }
    //截取位置
    if(n-n2 > 0)
        p = ssline.substr(n2,n-n2);
    // 移动n到下一个位置
    n++;
    return p;
}

int main(int argn, char** args){
    int i=0, n;
    ifstream exi;
    ename=args[1];
    string sline, word;

    // 打开文件流
    exi.open(ename.c_str(), ios::in);
    if(!exi){
        cout << "Couldn't open " << ename << endl;
        exit(-1);
    }

    // 读取文件内容
    while(getline(exi, sline)){
        i++;
        if(sline[0] == '@') //跳过 @开头的行
            continue;
        //cout << i << ": " << sline << endl;
        // 获取tab分隔的第0个
        n = 0;
		word = GetWordtab(n,sline);  //提取tab分隔的单词，要第一个
        cout << n << "|" << word << "|" << endl;
    }

    // 关闭文件流
    exi.close();
    cout << "读取文件结束" << endl;
    return 0;
}

========================================
c++内存管理 - 节点释放/销毁: malloc / calloc 和 free 配对(C风格的内存分配)，new / new[] 和 delete / delete[] 配对 (C++ 动态内存管理)
----------------------------------------
1.单节点释放

// 如果用 new 分配
TreeNode* node = new TreeNode{...};
delete node;  // 正确释放，会调用析构函数

// 如果用 malloc 分配：
TreeNode* node = (TreeNode*)malloc(sizeof(TreeNode));
free(node);    // 仅释放内存，不调用析构函数（可能导致 string 泄漏）



2. 递归释放整棵二叉树

void freeTree(TreeNode* root) {
    if (root == nullptr) return;
    freeTree(root->left);   // 递归释放左子树
    freeTree(root->right);  // 递归释放右子树
    delete root;            // 释放当前节点（若用 new 分配）
}




3. 示例

(1) c++ 推荐使用new: 将 malloc 替换为 new
// 原代码：
TreeNode* root = (TreeNode*)malloc(sizeof(TreeNode));
// 修改后：
TreeNode* root = new TreeNode{stu1, nullptr, nullptr};  // 初始化指针为 nullptr


(2) 释放时统一用 delete：
// 程序结束前释放整棵树
freeTree(root);  // 内部使用 delete




4.最佳实践建议
(1)优先使用 new/delete：确保对象构造/析构正确执行（尤其含 string 等非POD成员时）。
(2)避免混用：不要用 free 释放 new 分配的内存，反之亦然。
(3)使用智能指针（C++11+）：
#include <memory>
std::unique_ptr<TreeNode> root(new TreeNode{...});
// 自动释放内存，无需手动 delete

修正后的代码将不再有内存泄漏风险，且能正确处理 string 成员的析构。
